#include <iostream>
using namespace std;
int Rec(int n,int i)
{
    if(n<=0)
        return 1;
    return Rec(n-1,i)*i;
}

int main()
{
    int i;
    cout<<"enter valur: \n" ;
    cin>>i;
    int n;
    cout<<"Enter the power of n:\n";
    cin>>n;
    
    int ans1 = Rec(n,i);
    cout<<ans1;
}

#include <iostream>
using namespace std;
int fact(int n)
{
    if(n<=0)
        return 1;
    return fact(n-1)*n;
}

int main()
{
    int n;
    cout<<"Enter the power of n:\n";
    cin>>n;
    int s=fact(n);
    cout<<s;
}
