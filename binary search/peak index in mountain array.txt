#include <bits/stdc++.h>
using namespace std;
int peak(vector<int>arr){
    int s=0; 
    int e=arr.size()-1;
    
    while(s<=e){
        int mid=(s+e)/2;
        
        if(arr[mid]>arr[mid+1] && arr[mid]>arr[mid-1]){
            return mid;
            
        }
        else if(arr[mid]>arr[mid-1]){
            s=mid+1;
        }
        else if(arr[mid]<arr[mid-1]){
            e=mid-1;
        }
        
        // else{
        //     e=mid-1;
        // }
    }
    return -1;
}
int main(){
    vector<int>arr={2,4,12,14,15,18,5};
    int ans=peak(arr);
    cout<<ans;
    return 0;
}
