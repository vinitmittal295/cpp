7. Write a for loop to print the reverse of a given list.

#include<bits/stdc++.h>
using namespace std;
int reverse(int num,int rem,int sum)
{
    while(num>0){
        rem=num%10;
        sum=sum*10+rem;
        num=num/10;
    }
    return sum;
}
int main(){
    int num;
    int rem;
    int sum;
    int g;
    cout<<"enter any number";
    cin>>num;
    g=reverse(num,rem,sum);
    cout<<g;
}
