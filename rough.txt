#include <iostream>
using namespace std;
int armst(){
    int num;
    int digit;
    int sum=0;
    while(num>0){
        digit=num%10;
        sum=sum+digit*digit*digit;
        num=num/10;
        
    }
    return (sum); 
}

void armstrong(int start,int end ){
    for(int i=start; i<=end; i++){
        if(armst()){
            cout<<i<<"\n";
        }
    }
}

int main()
{
